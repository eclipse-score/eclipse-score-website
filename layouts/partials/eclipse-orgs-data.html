{{/* eclipse-orgs-data.html Fetches the Eclipse S-CORE member organizations from the Eclipse Foundation API at Hugo build time and embeds the result as window.__eclipseOrgs__ so that member-logos.js can use it without making a browser-side cross-origin request. If the remote fetch fails (network error, non-2xx status), this partial emits nothing and member-logos.js falls back to the static logos already in the HTML. */}} {{- $url := "https://membership.eclipse.org/api/projects/automotive.score/organizations" -}} {{- $remote := try (resources.GetRemote $url) -}} {{- with $remote -}} {{- with .Err -}} {{- /* On fetch failure, emit nothing. */ -}} {{- else -}} {{- with .Value -}} {{- $parsed := try (transform.Unmarshal .) -}} {{- with $parsed -}} {{- with .Err -}} {{- /* On parse failure, emit nothing. */ -}} {{- else -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
